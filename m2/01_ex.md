* CHAR(2) всегда занимает 2 байта
* VARCHAR(50) занимает только нужное место + 1-4 байта
* TEXT не имеет ограничений по длине

```sql
-- Создание таблицы с разными типами строк
CREATE TABLE string_examples (
    id SERIAL PRIMARY KEY,
    country_code CHAR(2),           -- всегда 2 символа: 'RU', 'US'
    name VARCHAR(50),               -- до 50 символов
    description TEXT                -- без ограничений
);

-- Вставка данных
INSERT INTO string_examples (country_code, name, description) VALUES
('RU', 'Россия', 'Российская Федерация - крупнейшая страна мира'),
('US', 'США', 'Соединенные Штаты Америки'),
('DE', 'Германия', 'Федеративная Республика Германия в центре Европы');

-- Проверка данных
SELECT * FROM string_examples;

-- Проверка длины строк
SELECT 
    country_code,
    LENGTH(country_code) as code_length,
    name,
    LENGTH(name) as name_length,
    LENGTH(description) as desc_length
FROM string_examples;

-- CHAR дополняется пробелами до фиксированной длины
SELECT 
    country_code,
    '|' || country_code || '|' as padded_code
FROM string_examples;
```
